function Gaussian2d(x0, y0, sizex=100, sizey=100, sigma=5)
    x = 0:sizex-1
    y = 0:sizey-1
    Y = [i for i in y, j in 1:length(x)]
    X = [j for i in 1:length(y), j in x]
    
    mask = exp.(-((X .- x0) .^2 + (Y .- y0) .^2) / (2.0*(sigma^2)))
    return mask ./ sum(mask)
end
